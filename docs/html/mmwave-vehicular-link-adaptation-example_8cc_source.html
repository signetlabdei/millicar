64 int
65 main (int argc, char *argv[])
66 {
67  double initialDistance = 10.0; // the initial distance between the two nodes in m
68  double finalDistance = 1000.0; // the final distance between the two nodes in m
69  double speed = 10.0; // the speed of the moving node in m/s
70  double ipi = 1.0; // the inter-packet interval in s
71  double frequency = 60e9; // the carrier frequency
72 
73  double endTime = (finalDistance - initialDistance) / speed; // time required for the simulation
74  uint16_t numOfTx = std::ceil (endTime / ipi); // the number of transmissions
75 
76  Config::SetDefault ("ns3::MmWavePhyMacCommon::CenterFreq", DoubleValue (frequency));
77  Config::SetDefault ("ns3::MmWaveVehicularPropagationLossModel::ChannelCondition", StringValue ("l"));
78 
79  // create the nodes
80  NodeContainer n;
81  n.Create (2);
82 
83  // create the mobility models
84  MobilityHelper mobility;
85  mobility.SetMobilityModel ("ns3::ConstantVelocityMobilityModel");
86  mobility.Install (n);
87 
88  n.Get (0)->GetObject<MobilityModel> ()->SetPosition (Vector (0, 0, 0));
89  n.Get (0)->GetObject<ConstantVelocityMobilityModel> ()->SetVelocity (Vector (0, 0, 0));
90 
91  n.Get (1)->GetObject<MobilityModel> ()->SetPosition (Vector (initialDistance, 0, 0));
92  n.Get (1)->GetObject<ConstantVelocityMobilityModel> ()->SetVelocity (Vector (speed, 0, 0));
93 
94  // create and configure the helper
95  Ptr<MmWaveVehicularHelper> helper = CreateObject<MmWaveVehicularHelper> ();
96  helper->SetNumerology (3);
97  helper->SetPropagationLossModelType ("ns3::MmWaveVehicularPropagationLossModel");
98  helper->SetSpectrumPropagationLossModelType ("ns3::MmWaveVehicularSpectrumPropagationLossModel");
99  NetDeviceContainer devs = helper->InstallMmWaveVehicularNetDevices (n);
100 
101  // Install the TCP/IP stack in the two nodes
102  InternetStackHelper internet;
103  internet.Install (n);
104 
105  Ipv4AddressHelper ipv4;
106  NS_LOG_INFO ("Assign IP Addresses.");
107  ipv4.SetBase ("10.1.1.0", "255.255.255.0");
108  Ipv4InterfaceContainer i = ipv4.Assign (devs);
109 
110  // Need to pair the devices in order to create a correspondence between transmitter and receiver
111  // and to populate the < IP addr, RNTI > map.
112  helper->PairDevices(devs);
113 
114  // Set the routing table
115  Ipv4StaticRoutingHelper ipv4RoutingHelper;
116  Ptr<Ipv4StaticRouting> staticRouting = ipv4RoutingHelper.GetStaticRouting (n.Get (0)->GetObject<Ipv4> ());
117  staticRouting->SetDefaultRoute (n.Get (1)->GetObject<Ipv4> ()->GetAddress (1, 0).GetLocal () , 2 );
118 
119  NS_LOG_DEBUG("IPv4 Address node 0: " << n.Get (0)->GetObject<Ipv4> ()->GetAddress (1, 0).GetLocal ());
120  NS_LOG_DEBUG("IPv4 Address node 1: " << n.Get (1)->GetObject<Ipv4> ()->GetAddress (1, 0).GetLocal ());
121 
122  // connect the SL scehduling callback
123  Ptr<MmWaveVehicularNetDevice> mmWaveNetDev = DynamicCast<MmWaveVehicularNetDevice> (devs.Get (0));
124  Ptr<MmWaveSidelinkMac> mac = mmWaveNetDev->GetMac ();
125  mac->TraceConnectWithoutContext ("SchedulingInfo", MakeCallback (&SlSchedulingCallbackSink));
126 
127  for (uint32_t i = 0; i < numOfTx; i++)
128  {
129  // node 0 tx to node 1
130  Ptr<Packet> p = Create<Packet> (100);
131  Ipv4Header iph;
132  iph.SetDestination (n.Get (1)->GetObject<Ipv4> ()->GetAddress (1, 0).GetLocal ());
133  p->AddHeader (iph);
134  Simulator::Schedule (MilliSeconds (1000 * i * ipi), &Tx, p, devs.Get (0));
135 
136  // node 1 tx to node 0
137  Ptr<Packet> p1 = Create<Packet> (100);
138  Ipv4Header iph1;
139  iph1.SetDestination (n.Get (0)->GetObject<Ipv4> ()->GetAddress (1, 0).GetLocal ());
140  p1->AddHeader (iph1);
141  Simulator::Schedule (MilliSeconds (1000 * i * ipi), &Tx, p1, devs.Get (1));
142  }
143 
144  std::ofstream myfile;
145  myfile.open ("mmwave-vehicular-link-adaptation-example-out.txt", std::ios::out | std::ios::trunc);
146  myfile << "Time MCS" << std::endl;
147  myfile.close();
148 
149  Simulator::Stop (Seconds (endTime));
150  Simulator::Run ();
151  Simulator::Destroy ();
152 }
